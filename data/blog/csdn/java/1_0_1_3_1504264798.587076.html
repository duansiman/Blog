/**
  * 批处理保存信息
  */
 public void batchsave(List<T> entity) {
  Session session = getSession();
  T t = null; // 创建药品对象
  // 循环获取药品对象
  for (int i = 0; i < entity.size(); i++) {
   t = (T) entity.get(i); // 获取药品
   session.save(t); // 保存药品对象
   // 批插入的对象立即写入数据库并释放内存
   if (i % 50 == 0) {
    session.flush();
    session.clear();
   }
  }
  session.flush();
  session.clear();
 }
