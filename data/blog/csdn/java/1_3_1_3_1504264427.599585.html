/**
	 * 22.验证汉字 
	 * 表达式 ^[\u4e00-\u9fa5]{0,}$ 
	 * 描述 只能汉字 
	 * 匹配的例子 清清月儿 
	 */
	@Test
	public void a1() {
		Scanner sc = new Scanner(System.in);
		String input = sc.nextLine();
		String regex = "^[\\u4e00-\\u9fa5]*$";
		Matcher m = Pattern.compile(regex).matcher(input);
		System.out.println(m.find());
		sc.close();
	}
