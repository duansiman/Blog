       日期时间函数

\n

       常用的日期时间函数：

\n


\n

\n

       (1)当前日期和时间以及当前日期和当前时间

\n

       SELECT NOW();

\n       SELECT CURDATE();
\n \n

       SELECT CURTIME();

\n


\n

\n

       (2)日期变化

\n

       SELECT DATE_ADD('2016-5-1',INTERVAL 23 DAY);

\n       SELECT DATE_ADD('2016-5-1',INTERVAL -23 DAY);
\n \n

       SELECT DATE_ADD('2016-5-1',INTERVAL 3 WEEK);

\n


\n

\n

       (3)日期差值和日期格式化

\n

       SELECT DATEDIFF('2016-5-1','2016-5-23');

\n       SELECT DATEDIFF('2016-5-15','2016-5-3');
\n \n

       SELECT DATE_FORMAT('2016-5-2','%m/%d/%Y');
