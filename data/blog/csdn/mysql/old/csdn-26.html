\n 5.修改表

\n
     5.1 添加字段
\n
      sql语句：alter table 表名 \n add column 新添字段名 类型;
\n
\n
\n
  1. alter table student add column sgender varchar(2);
\n
\n

\n
 5.2删除字段
\n
\n
\n
   sql语句：alter table 表名 drop column \n 字段名;
\n
\n
\n
\n
  1. alter table student drop column sgender;
\n
\n

\n
\n
\n
\n
 5.3修改字段类型
\n
\n
     sql语句：alter \n table 表名 modify column 字段名 修改后的新类型;
\n
\n
\n
  1. alter table student modify column remark varchar(100);
\n
\n

\n
\n
\n
\n
 5.4修改字段名称
\n
     sql语句：alter \n table 表名 change column 字段名 修改后的字段名 修改后的新类型;

\n
\n
\n
  1. alter table student change column gender sgender varchar(2);
