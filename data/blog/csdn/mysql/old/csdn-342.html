service mysql start出错，mysql启动不了，解决mysql: unrecognized service错误的方法如下：

\n

[root@ctohome.com ~]# service mysql start
\nmysql: unrecognized service
\n[root@ctohome.com ~]# service mysql restart
\nmysql: unrecognized service
\n
\n[root@ctohome.com ~]# rpm -q mysql 查询发现mysql已经正常安装
\nmysql-5.1.52-jason.1

\n

[root@ctohome.com ~]# /etc/rc.d/init.d/mysqld start 直接启动没问题
\nStarting mysqld: [ OK ]

\n


\n[root@ctohome.com ~]# ls /etc/rc.d/init.d/mysqld -l
\n-rwxr-xr-x 1 root root 5509 Dec 18 02:31 /etc/rc.d/init.d/mysqld

\n

[root@ctohome.com ~]# chkconfig mysqld on 设置mysql开机启动

\n

[root@ctohome.com ~]# chmod 755 /etc/rc.d/init.d/mysqld 修改mysqld执行权限

\n

[root@ctohome.com ~]# service mysqld start 搞定
\nStarting mysqld: [ OK ]
\n[root@ctohome.com ~]# service mysqld start
\nStarting mysqld: [ OK ]
\n[root@ctohome.com ~]# service mysqld status
\nmysqld (pid 9487) is running...
