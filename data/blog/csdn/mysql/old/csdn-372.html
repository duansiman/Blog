3.禁用或卸载iptables，这里我们将iptables禁用：

\n

1) 关闭防火墙-----service iptables stop
\n2)启动防火墙-----service iptables start
\n3)重启防火墙-----service iptables restart
\n4) 查看防火墙状态--service iptables status
\n5) 永久关闭防火墙--chkconfig iptables off
\n6) 永久关闭后启用--chkconfig iptables on

三、安装DRBD 8.4.2（主备节点配置相同）

\n

1.主备节点安装drbd编译环境：
\n

\n

\n
# yum -y install gcc flex wget make kernel-devel kernel-headers Git libxslt rpm-build automake autoconf
\n

\n

\n2.下载DRBD安装文件

\n

\n

\n# wget http://oss.linbit.com/drbd/8.4/drbd-8.4.2.tar.gz

\n

3.解压并安装DRBD文件

\n

\n
#tar -zvxf drbd-8.4.2.tar.gz\n# cd drbd-8.4.2\ndrbd-8.4.2]# ./configure --prefix=/usr/local/drbd --with-km\ndrbd-8.4.2]# make KDIR=/usr/src/kernels/`uname -r`/\ndrbd-8.4.2]# make install
\n

\n

4.安装配置DRBD驱动模块

\n

\n
drbd-8.4.2]# cd drbd\ndrbd-8.4.2]# make clean\ndrbd-8.4.2]# make KDIR=/usr/src/kernels/`uname -r`/\ndrbd-8.4.2]# cp drbd.ko /lib/modules/`uname -r`/kernel/lib/\ndrbd-8.4.2]# depmod #创建模块依赖关系的列表\ndrbd-8.4.2]# cp /usr/local/drbd/etc/rc.d/init.d/drbd /etc/rc.d/init.d/\ndrbd-8.4.2]# chkconfig --add drbd\ndrbd-8.4.2]# chkconfig drbd on
\n

\n

四、定义DRBD 配置文件 （主备节点配置相同）

\n

\n
~]# cp /usr/local/drbd/etc/drbd.d/global_common.conf /usr/local/drbd/etc/drbd.d/global_common-bak.conf\n~]# vi /usr/local/drbd/etc/drbd.d/global_common.conf
 global {\n usage-count yes; # 是否参加DRBD使用者统计.默认是yes\n # minor-count dialog-refresh disable-ip-verification\n}\n\ncommon {\n handlers {\n pri-on-incon-degr "/usr/lib/drbd/notify-pri-on-incon-degr.sh; /usr/lib/drbd/notify-emergency-reboot.sh; echo b > /proc/sysrq-trigger ; reboot -f";\n pri-lost-after-sb "/usr/lib/drbd/notify-pri-lost-after-sb.sh; /usr/lib/drbd/notify-emergency-reboot.sh; echo b > /proc/sysrq-trigger ; reboot -f";\n local-io-error "/usr/lib/drbd/notify-io-error.sh; /usr/lib/drbd/notify-emergency-shutdown.sh; echo o > /proc/sysrq-trigger ; halt -f";\n }\n\n startup {\n wfc-timeout 30;\n degr-wfc-timeout 30;\n outdated-wfc-timeout 30;\n }\n\n options {\n # cpu-mask on-no-data-accessible\n }\n\n disk {\n on-io-error detach;\n fencing resource-and-stonith;\n resync-rate 50M; # 设置主备节点同步时的网络速率最大值,单位是字节.\n }\n\n net {\n protocol C; # 使用协议C.表示收到远程主机的写入确认后,则认为写入完成.\n cram-hmac-alg sha1; # 设置主备机之间通信使用的信息算法.\n shared-secret "123456";\n }\n}
~]# vi /usr/local/drbd/etc/drbd.d/Redis.res
 resource redis{ \n on master { \n device /dev/drbd1; #逻辑设备的路径 \n disk /dev/sdb; #物理设备 \n address 192.168.0.152:7788; #IP和监听端口\n meta-disk internal; \n } \n on slave { \n device /dev/drbd1; \n disk /dev/sdb; \n address 192.168.0.153:7788; \n meta-disk internal; \n } \n}
