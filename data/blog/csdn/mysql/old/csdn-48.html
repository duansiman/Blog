\n

\n

mysql> SELECT DATE_ADD('2016-3-12',INTERVAL 365 DAY);\n+----------------------------------------+\n| DATE_ADD('2016-3-12',INTERVAL 365 DAY) |\n+----------------------------------------+\n| 2017-03-12 |\n+----------------------------------------+\n1 row in set (0.00 sec)\n\nmysql> SELECT DATE_ADD('2016-3-12',INTERVAL -365 DAY);\n+-----------------------------------------+\n| DATE_ADD('2016-3-12',INTERVAL -365 DAY) |\n+-----------------------------------------+\n| 2015-03-13 |\n+-----------------------------------------+\n1 row in set (0.00 sec)\n\nmysql> SELECT DATE_ADD('2016-3-12',INTERVAL 1 YEAR);\n+---------------------------------------+\n| DATE_ADD('2016-3-12',INTERVAL 1 YEAR) |\n+---------------------------------------+\n| 2017-03-12 |\n+---------------------------------------+\n1 row in set (0.00 sec)\n\nmysql> SELECT DATE_ADD('2016-3-12',INTERVAL 3 WEEK);\n+---------------------------------------+\n| DATE_ADD('2016-3-12',INTERVAL 3 WEEK) |\n+---------------------------------------+\n| 2016-04-02 |\n+---------------------------------------+\n1 row in set (0.00 sec)\n

\n可以使用的格式有：
\n
\n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n \n
\n格式 \n描述
\n%a \n缩写星期名
\n%b \n缩写月名
\n%c \n月，数值
\n%D \n带有英文前缀的月中的天
\n%d \n月的天，数值(00-31)
\n%e \n月的天，数值(0-31)
\n%f \n微秒
\n%H \n小时 (00-23)
\n%h \n小时 (01-12)
\n%I \n小时 (01-12)
\n%i \n分钟，数值(00-59)
\n%j \n年的天 (001-366)
\n%k \n小时 (0-23)
\n%l \n小时 (1-12)
\n%M \n月名
\n%m \n月，数值(00-12)
\n%p \nAM 或 PM
\n%r \n时间，12-小时（hh:mm:ss AM 或 PM）
\n%S \n秒(00-59)
\n%s \n秒(00-59)
\n%T \n时间, 24-小时 (hh:mm:ss)
\n%U \n周 (00-53) 星期日是一周的第一天
\n%u \n周 (00-53) 星期一是一周的第一天
\n%V \n周 (01-53) 星期日是一周的第一天，与 %X 使用
\n%v \n周 (01-53) 星期一是一周的第一天，与 %x 使用
\n%W \n星期名
\n%w \n周的天 （0=星期日, 6=星期六）
\n%X \n年，其中的星期日是周的第一天，4 位，与 %V 使用
\n%x \n年，其中的星期一是周的第一天，4 位，与 %v 使用
\n%Y \n年，4 位
\n%y \n年，2 位
