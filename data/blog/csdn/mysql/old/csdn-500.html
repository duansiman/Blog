\n连接 MySQL

\n

\n

\n

\n修改密码

\n
\n
\n格式：mysqladmin -u用户名 -p旧密码 password 新密码
\n
\n1．例1：给root加个密码ab12。首先在DOS下进入目录mysqlbin，然后键入以下命令
\n
\nmysqladmin -uroot -password ab12
\n
\n注：因为开始时root没有密码，所以-p旧密码一项就可以省略了。
\n
\n2．例2：再将root的密码改为djg345。
\n
\nmysqladmin -uroot -pab12 password djg345

\ndrop database if exists school; //如果存在SCHOOL则删除

\ncreate database school; //建立库SCHOOL
\n
\nuse school; //打开库SCHOOL
\n
\ncreate table teacher //建立表TEACHER
\n
\n(
\n
\nid int(3) auto_increment not null primary key,
\n
\nname char(10) not null,
\n
\naddress varchar(50) default ‘深圳',
\n
\nyear date
\n
\n); //建表结束

\n
\nmysql -uroot -p密码 < c:\\school.sql
