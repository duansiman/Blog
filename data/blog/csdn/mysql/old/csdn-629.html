1. 远程登录Mysql服务器：

\n

\n
mysql -u用户名 -p -hIP地址 -P端口号
然后输入密码， 即可登录到远程服务器 \n

\n


\n

\n


\n

\n

2. 显示／查看数据库

\n

\n
show databases;\n

\n
\n

\n

3. 选择数据库

\n

\n
use 数据库名;

\n
\n

\n

4. 显示／查看数据表

\n

\n
show tables;

\n
\n

\n

5. 查看具体表结构

\n

\n
describe 具体表名;

\n
\n

6. 事务提交SQL

\n

\n
START TRANSACTION;\nINSERT INTO table VALUES("AA","text","0","2015-07-23 10:41:57","2015-07-23 10:41:57");\n\nINSERT INTO table VALUES("BB","text","100","2015-07-23 10:41:57","2015-07-23 10:41:57");\n\nCOMMIT;

\n


\n

7. 修改字段大小

\n

\n
ALTER TABLE table_name MODIFY COLUMN 字段名 varchar(255);

\n
\n

8. 增加字段

\n

\n
ALTER TABLE table_name ADD field_name field_type DEFAULT default_value;

\n
\n

9. 查看索引

\n

\n
show index from table_name;

\n
\n

10.  查看所有字段

\n

show full columns from table_name;
