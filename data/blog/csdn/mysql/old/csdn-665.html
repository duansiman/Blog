4、创建存储过程

\n

\n

\n
delimiter //\n \ncreate procedure querystudent(\n in in_id int , #0- Character ID 1-digital id \n #\n out out_ret int, # Returns the result \n out out_name varchar(20), # First name \n out out_age int # Age \n)\nlabel_a:begin\n declare v_name varchar(20) ;\n declare v_age tinyint ;\n # Parameters determine \n if (in_id<=0) then\n set out_ret=-1; #id error\n leave label_a;\n end if;\n \n SELECT name,age into v_name,v_age from student where't found\n leave label_a;\n end if; \n \n set out_ret=0;\n set out_name=v_name;\n set out_age=v_age;\nend;\n//\ndelimiter ;

\n

\n

\n

5.1、Mysql 的句柄初始化

\n

\n
if(!mysql_init(&mysql))\n {\n printf("mysql_init failed!\\n");\n return 0;\n }

\n

5.2、连接到MySQL数据库

\n

\n

\n
//login or connect\n if(!mysql_real_connect(&mysql,"localhost","root","","billingdb",0,NULL,CLIENT_MULTI_STATEMENTS))\n {\n printf("mysql_real_connect() failed!\\n");\n mysql_close(&mysql);\n return 0;\n }

\n

5.3、调用存储过程

\n

\n

\n
//call\n strcpy(query,"call querystudent (1,@ret,@ out_name,@ out_age)");\n printf("query sql=[%s]\\n",query);\n ret= mysql_real_query(&mysql,query,(unsigned int)strlen(query));

\n

5.4、查询返回值及输出参数值

\n

\n

\n
mysql_query(&mysql, "SELECT @ret,@ out_name,@ out_age ");\n \n //get result\n if (ret)\n {\n printf("Error exec query: %s\\n",mysql_error(&mysql));\n }\n else\n {\n printf("[%s] exec...\\n", query);\n }\n \n results = mysql_store_result(&mysql);
